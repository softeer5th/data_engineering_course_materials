#!/bin/bash

# docker 설치
sudo yum update -y

sudo yum install docker -y

sudo service docker start

sudo usermod -aG docker ec2-user

# Login aws
aws configure set aws_access_key_id <aws_access_key_id>

aws configure set aws_secret_access_key <aws_secret_access_key>

aws configure set default.region <region>

# ecr 연결
aws ecr get-login-password --region ap-northeast-2 | docker login --username AWS --password-stdin 619071320336.dkr.ecr.ap-northeast-2.amazonaws.com

# 도커 pull run
docker pull 619071320336.dkr.ecr.ap-northeast-2.amazonaws.com/softeer/w2
